def login_from_token
  return if request.headers['HTTP_AUTH_TOKEN'].blank?

  if verify_auth_token
    @access_token = AccessToken.find_by token: request.headers['HTTP_AUTH_TOKEN']
  end
  if @access_token
    @current_user ||= @access_token.user
  end
end
