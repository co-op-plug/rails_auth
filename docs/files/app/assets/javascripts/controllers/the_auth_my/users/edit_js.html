attachToInput({

input: 'proxy',
fileInput: 'file_input',
uploadUrl: '/my/user',
uploadFieldName: 'user[avatar]',

  token: '',
  _method: 'patch'
},

  'Accept': 'application/json',
  'X-CSRF-Token': $('meta[name="csrf-token"]').attr('content')
},

  $('#avatar_image').attr('src', filename);
}

});
